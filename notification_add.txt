<script>
document.addEventListener(" DOMContentLoaded\, function() {
 console.log(\Notification Enhancement Active\);
 const panel = document.getElementById(\incident-notifications-dashboard\);
 if (panel) {
 panel.style.display = \block\;
 panel.style.visibility = \visible\;
 panel.style.opacity = \1\;
 panel.classList.remove(\d-none\);
 }
 loadDashboardNotifications();
 setInterval(loadDashboardNotifications, 15000);
});

function loadDashboardNotifications() {
 fetch(\/api/get_pending_assignments\)
 .then(response => response.json())
 .then(data => {
 if (data.success && data.assignments && data.assignments.length > 0) {
 displayDashboardNotifications(data.assignments);
 }
 })
 .catch(error => console.error(\Error:\, error));
}

function displayDashboardNotifications(assignments) {
 const panel = document.getElementById(\incident-notifications-dashboard\);
 const countSpan = document.getElementById(\notifications-count\);
 const summaryDiv = document.getElementById(\notifications-summary\);
 
 if (panel && countSpan && summaryDiv) {
 panel.style.display = \block\;
 countSpan.textContent = assignments.length;
 summaryDiv.innerHTML = \<strong>\ + assignments.length + " pending assignments</strong>\;
    }
}

function toggleNotificationsList() {
    const listDiv = document.getElementById(\notifications-list-dashboard\);
    if (listDiv) {
        listDiv.style.display = listDiv.style.display === \none\ ? \block\ : \none\;
    }
}
</script>
